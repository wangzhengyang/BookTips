#获取当前时间并生成唯一文件名
```bash
#!/bin/bash
today=$(date +%y%m%d)
ls /usr/bin -al > log.$today
```

# 循环处理文件数据
```bash
#!/bin/bash
IFS.OLD=$IFS
IFS=$'\n'
for entry in $(cat /etc/passwd)
do
    echo "Values in $entry -"
    IFS=:
    for value in $entry
    do
        echo "   $value"
    done
done
```
#查找可执行文件
```bash
#!/bin/bash
IFS:
for folder in $PATH
do
    echo "$folder:"
    for file in $folder/*
    do
        if [ -x $file ]
        then
            echo " $file"
        fi
    done
done
```

#创建多个用户账号
```bash
#!/bin/bash
input="users.csv"
while IFS=',' read -r userid name
do
    echo "adding $userid"
    useradd -c "$name" -m $userid
done < "$input"
```
